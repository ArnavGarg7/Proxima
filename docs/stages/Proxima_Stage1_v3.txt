"""
ModelRegistry: the single source of truth for all AI model configuration.
NO other file in the codebase may reference model strings directly.
All AI calls go through this service.
"""

from proxima.models.routing import RegisteredModel, ModelRoutingRule
from proxima.database import get_db
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
import structlog

logger = structlog.get_logger()

class ModelRegistry:

    async def get_default_generation(self, db: AsyncSession) -> RegisteredModel:
        """Returns the active default generation model."""
        stmt = select(RegisteredModel).where(
            RegisteredModel.model_type == 'generation',
            RegisteredModel.is_default_generation == True,
            RegisteredModel.is_active == True,
        )
        result = await db.execute(stmt)
        model = result.scalar_one_or_none()
        if not model:
            raise RuntimeError("No active default generation model found in registry.")
        return model

    async def get_default_embedding(self, db: AsyncSession) -> RegisteredModel:
        """Returns the active default embedding model."""
        stmt = select(RegisteredModel).where(
            RegisteredModel.model_type == 'embedding',
            RegisteredModel.is_default_embedding == True,
            RegisteredModel.is_active == True,
        )
        result = await db.execute(stmt)
        model = result.scalar_one_or_none()
        if not model:
            raise RuntimeError("No active default embedding model found in registry.")
        return model

    async def get_for_task(self, task_class: str, domain: str | None,
                            db: AsyncSession) -> RegisteredModel:
        """Returns the best registered model for a given task + domain."""
        # Try specific domain match first
        if domain:
            stmt = select(ModelRoutingRule).join(
                RegisteredModel, ModelRoutingRule.model_id == RegisteredModel.model_id
            ).where(
                ModelRoutingRule.task_class == task_class,
                ModelRoutingRule.domain == domain,
                ModelRoutingRule.is_active == True,
                RegisteredModel.is_active == True,
            ).order_by(ModelRoutingRule.priority)
            result = await db.execute(stmt)
            rule = result.scalar_one_or_none()
            if rule:
                return await db.get(RegisteredModel, rule.model_id)

        # Fall back to task-level (domain=NULL) rule
        stmt = select(ModelRoutingRule).where(
            ModelRoutingRule.task_class == task_class,
            ModelRoutingRule.domain == None,
            ModelRoutingRule.is_active == True,
        ).order_by(ModelRoutingRule.priority)
        result = await db.execute(stmt)
        rule = result.scalar_one_or_none()
        if rule:
            return await db.get(RegisteredModel, rule.model_id)

        # Ultimate fallback: default generation model
        return await self.get_default_generation(db)

    async def build_generation_client(self, model: RegisteredModel):
        """
        Returns a provider-appropriate generation client.
        Application code uses this client — never imports provider SDKs directly.
        """
        if model.provider == 'google':
            import google.generativeai as genai
            from proxima.config import settings
            genai.configure(api_key=settings.gemini_api_key)
            return genai.GenerativeModel(model.model_id)

        elif model.provider == 'openai':
            from openai import AsyncOpenAI
            from proxima.config import settings
            return AsyncOpenAI(api_key=settings.openai_api_key)

        elif model.provider == 'anthropic':
            import anthropic
            from proxima.config import settings
            return anthropic.AsyncAnthropic(api_key=settings.anthropic_api_key)

        raise ValueError(f"Unknown provider: {model.provider}")

    async def build_embedding_client(self, model: RegisteredModel):
        """Returns a provider-appropriate embedding client."""
        if model.provider == 'google':
            import google.generativeai as genai
            from proxima.config import settings
            genai.configure(api_key=settings.gemini_api_key)
            return genai  # embed_content called directly on module
        raise ValueError(f"Unknown embedding provider: {model.provider}")

    async def stream_completion(
        self, model: RegisteredModel, system_prompt: str,
        user_message: str, temperature: float, max_tokens: int
    ):
        """
        Unified streaming interface across all providers.
        Yields (token_text: str) for each token.
        """
        client = await self.build_generation_client(model)

        if model.provider == 'google':
            response = await client.generate_content_async(
                user_message,
                stream=True,
                generation_config={
                    'temperature': temperature,
                    'max_output_tokens': max_tokens,
                },
                system_instruction=system_prompt,
            )
            async for chunk in response:
                if chunk.text:
                    yield chunk.text

        elif model.provider == 'openai':
            stream = await client.chat.completions.create(
                model=model.model_id,
                messages=[
                    {"role": "system", "content": system_prompt},
                    {"role": "user",   "content": user_message},
                ],
                temperature=temperature,
                max_tokens=max_tokens,
                stream=True,
            )
            async for chunk in stream:
                delta = chunk.choices[0].delta.content
                if delta:
                    yield delta

        elif model.provider == 'anthropic':
            async with client.messages.stream(
                model=model.model_id,
                system=system_prompt,
                messages=[{"role": "user", "content": user_message}],
                temperature=temperature,
                max_tokens=max_tokens,
            ) as stream:
                async for text in stream.text_stream:
                    yield text

model_registry = ModelRegistry()  # singleton

BACKEND — auth_router.py — COMPLETE OAuth with PKCE
═══════════════════════════════════════════════════════════════════════

GET /api/auth/google
  1. Generate PKCE:
     code_verifier = secrets.token_urlsafe(64)  # 64-char random
     code_challenge = base64url(sha256(code_verifier.encode()))
     # Store: await redis.setex(f"pkce:{state}", 600, code_verifier)
  2. Generate state: state = secrets.token_urlsafe(32)
     await redis.setex(f"oauth_state:{state}", 600, "valid")
  3. Build Google OAuth URL:
     params = {
       "client_id":             settings.google_client_id,
       "redirect_uri":          settings.google_redirect_uri,
       "response_type":         "code",
       "scope":                 "openid email profile",
       "state":                 state,
       "code_challenge":        code_challenge,
       "code_challenge_method": "S256",
       "access_type":           "offline",
     }
  4. Return RedirectResponse to Google OAuth URL

GET /api/auth/callback?code=...&state=...
  1. Validate state: check Redis key "oauth_state:{state}" exists, then delete
  2. Retrieve code_verifier from Redis: "pkce:{state}"
  3. Exchange code for tokens (include code_verifier in token request):
     POST https://oauth2.googleapis.com/token with code_verifier
  4. Fetch Google user info
  5. Upsert user in DB (set role='user' on create)
  6. Issue RS256 JWT tokens:
     access_token  = create_access_token(user_id, email, plan, role)
     refresh_token = create_refresh_token(user_id)
  7. Store refresh token in Redis:
     await redis.setex(f"refresh:{user_id}", 7*24*3600, refresh_token_hash)
  8. Set cookies:
     access_token:  HttpOnly, Secure (prod), SameSite=Lax, Max-Age=900
     refresh_token: HttpOnly, Secure, SameSite=Lax, Max-Age=604800
  9. Return RedirectResponse to FRONTEND_URL/dashboard

POST /api/auth/refresh
  1. Read refresh_token cookie
  2. Verify JWT signature
  3. Validate against Redis: f"refresh:{user_id}" — reject if not found
  4. DELETE old refresh token from Redis (rotation)
  5. Issue new access_token AND new refresh_token
  6. Store new refresh_token hash in Redis
  7. Set both new cookies
  8. Return { ok: true }

POST /api/auth/logout
  1. Read access_token cookie, extract user_id
  2. DELETE refresh token from Redis: f"refresh:{user_id}" (revocation)
  3. Clear both cookies (set Max-Age=0)
  4. Return { ok: true }

  3. Fetch system prompt from prompt_versions where prompt_key='system.base' and is_active=TRUE
     If not found: fallback = "You are Proxima. Complete the text professionally."
  4. Extract context: immediate_context = text[max(0, cursor_position-2000):cursor_position]
  5. Stream via ModelRegistry:
     async def generate():
         full = ''
         async for token in model_registry.stream_completion(
             model, system_prompt, immediate_context, temperature, max_tokens
         ):
             full += token
             yield f"data: {json.dumps({'token':token,'done':False})}\n\n"
         yield f"data: {json.dumps({'token':'','done':True})}\n\n"
         asyncio.create_task(log_and_update(full, model, user, db))
     return StreamingResponse(generate(), media_type="text/event-stream")
