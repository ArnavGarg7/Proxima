"""
Upload security pipeline — PRD §12.2.
All file uploads MUST pass this pipeline before any processing.
"""

import magic
from pathlib import Path

# Allowed MIME types and their magic byte signatures
ALLOWED_TYPES = {
    'application/pdf':          b'%PDF',
    'application/vnd.openxmlformats-officedocument.wordprocessingml.document':
                                bytes([0x50, 0x4B, 0x03, 0x04]),  # ZIP (DOCX is a ZIP)
    'text/plain':               None,  # text files have no magic bytes
}

MAX_FILE_SIZE  = 20 * 1024 * 1024   # 20 MB
ARCHIVE_TYPES  = {'.zip', '.tar', '.gz', '.7z', '.rar', '.bz2'}

class UploadSecurityError(ValueError):
    pass

class UploadSecurityPipeline:

    def validate(self, file_bytes: bytes, filename: str, declared_mime: str) -> str:
        """
        Run all 5 validation steps.
        Returns validated MIME type.
        Raises UploadSecurityError on any failure.
        """
        mime = self._step1_mime_validation(declared_mime)
        self._step2_magic_byte_validation(file_bytes, mime)
        self._step3_size_validation(file_bytes)
        self._step4_archive_bomb_protection(filename)
        self._step5_malware_scan_hook(file_bytes, filename)
        return mime

    def _step1_mime_validation(self, declared_mime: str) -> str:
        """Step 1: MIME type must be in allowed list."""
        clean_mime = declared_mime.split(';')[0].strip().lower()
        if clean_mime not in ALLOWED_TYPES and clean_mime != 'text/plain':
            raise UploadSecurityError(
                f"File type '{clean_mime}' is not permitted. "
                "Allowed: PDF, DOCX, TXT"
            )
        return clean_mime

    def _step2_magic_byte_validation(self, file_bytes: bytes, mime: str) -> None:
        """Step 2: Verify file content matches declared MIME via magic bytes."""
        expected_sig = ALLOWED_TYPES.get(mime)
        if expected_sig is None:
            return  # text/plain has no magic bytes — skip
        if not file_bytes[:len(expected_sig)] == expected_sig:
            raise UploadSecurityError(
                "File content does not match declared type. Upload rejected."
            )

    def _step3_size_validation(self, file_bytes: bytes) -> None:
        """Step 3: File must be under 20 MB."""
        if len(file_bytes) > MAX_FILE_SIZE:
            raise UploadSecurityError(
                f"File exceeds maximum size of 20 MB "
                f"({len(file_bytes) / 1024 / 1024:.1f} MB uploaded)."
            )

    def _step4_archive_bomb_protection(self, filename: str) -> None:
        """Step 4: Reject archive file extensions regardless of MIME."""
        ext = Path(filename).suffix.lower()
        if ext in ARCHIVE_TYPES:
            raise UploadSecurityError(
                f"Archive files ({ext}) are not permitted."
            )

    def _step5_malware_scan_hook(self, file_bytes: bytes, filename: str) -> None:
        """
        Step 5: Malware scan hook.
        v1.0: stub always returns clean (True).
        Future: plug in ClamAV or VirusTotal API here.
        """
        is_clean = self._scan(file_bytes, filename)
        if not is_clean:
            raise UploadSecurityError("File failed malware scan. Upload rejected.")

    def _scan(self, file_bytes: bytes, filename: str) -> bool:
        """Pluggable scan. Override for ClamAV integration."""
        return True  # stub — always clean in v1.0

upload_security = UploadSecurityPipeline()

"""
Quality Heuristic Engine (QHE).

External documentation calls this "Confidence Scoring" or "Confidence Audit".
Internally it is always "Quality Heuristic Engine" or "QHE".

Thresholds (PRD §6 — canonical, do not copy):
  80-100 → high     (#22c55e)
  65-79  → amber    (#f59e0b)
  50-64  → low      (#f97316)
  0-49   → critical (#ef4444)
"""

from scipy.special import kl_div
import numpy as np

# Domain-specific weighting coefficients (PRD §7.1)
QHE_COEFFICIENTS = {
    'legal':   {'T': 0.35, 'S': 0.30, 'D': 0.20, 'C': 0.15},
    'medical': {'T': 0.30, 'S': 0.25, 'D': 0.25, 'C': 0.20},
    'code':    {'T': 0.20, 'S': 0.40, 'D': 0.25, 'C': 0.15},
    'default': {'T': 0.25, 'S': 0.25, 'D': 0.25, 'C': 0.25},
}

# Vocabulary sets for Terminology Accuracy dimension
LEGAL_TERMS   = {'whereas','heretofore','notwithstanding','indemnify','hereinafter',
                 'pursuant','covenant','liability','jurisdiction','breach','shall',
                 'hereby','thereof','therein','hereof','hereunder','termination'}
MEDICAL_TERMS = {'diagnosis','prognosis','patient','symptoms','medication','clinical',
                 'etiology','dosage','adverse','vital','tachycardia','hypertension',
                 'CBC','BMP','EKG','MRI','troponin','hemoglobin','analgesic'}
CODE_TERMS    = {'function','class','import','const','let','var','def','return',
                 'async','await','null','undefined','interface','implements'}

TERM_SETS = {'legal': LEGAL_TERMS, 'medical': MEDICAL_TERMS, 'code': CODE_TERMS}

# Structural patterns per domain
import re
LEGAL_STRUCTURAL   = [r'\b(?:whereas|heretofore|pursuant|covenant)\b',
                      r'\d+\.\d+', r'\([a-z]\)\s', r'Article\s+[IVX]+']
MEDICAL_STRUCTURAL = [r'\bS:', r'\bO:', r'\bA:', r'\bP:', r'Chief Complaint:',
                      r'\b(?:mg|mmHg|ml|kg)\b', r'\b(?:BID|TID|QID|PRN)\b']
CODE_STRUCTURAL    = [r'=>', r'\{[^}]{1,80}\}', r'def\s+\w+\s*\(', r'function\s+\w+']
STRUCT_PATTERNS    = {'legal': LEGAL_STRUCTURAL, 'medical': MEDICAL_STRUCTURAL, 'code': CODE_STRUCTURAL}

class QualityHeuristicEngine:

    def score_segment(self, segment: str, domain_weights: dict, context: str) -> dict:
        seg_lower = segment.lower()
        seg_words = set(seg_lower.split())
        dominant  = max(domain_weights, key=domain_weights.get)
        coeffs    = QHE_COEFFICIENTS.get(dominant, QHE_COEFFICIENTS['default'])

        # T — Terminology Accuracy
        expected  = TERM_SETS.get(dominant, set())
        other     = set().union(*[v for k, v in TERM_SETS.items() if k != dominant])
        correct   = len(seg_words & expected)
        wrong     = len(seg_words & other)
        T = min(100.0, max(0.0, 60.0 + correct * 15.0 - wrong * 10.0)) if expected else 70.0

        # S — Structural Validity
        patterns  = STRUCT_PATTERNS.get(dominant, [])
        matches   = sum(1 for p in patterns if re.search(p, segment, re.IGNORECASE))
        length_penalty = max(0, 20 - len(segment.split())) if len(segment.split()) < 20 else 0
        S = min(100.0, max(0.0, 55.0 + matches * 10.0 - length_penalty))

        # D — Domain Consistency (KL divergence)
        from proxima.services.domain_detector import domain_detector
        seg_weights = domain_detector.detect(segment)
        domains = ['code', 'medical', 'legal']
        eps = 1e-8
        p = np.array([domain_weights.get(d, 0) + eps for d in domains])
        q = np.array([seg_weights.get(d, 0)     + eps for d in domains])
        p /= p.sum(); q /= q.sum()
        kl_pq = float(np.sum(kl_div(p, q)))
        kl_qp = float(np.sum(kl_div(q, p)))
        sym_kl = (kl_pq + kl_qp) / 2.0
        D = min(100.0, max(0.0, 100.0 - sym_kl * 50.0))

        # C — Contextual Coherence
        context_words = set(context.lower().split())
        seg_content   = {w for w in seg_words if len(w) > 4}
        shared        = len(context_words & seg_content)
        boundary_bonus = 10.0 if context.rstrip().endswith(('.', '!', '?', ';')) else 0.0
        C = min(100.0, max(0.0, 50.0 + shared * 4.0 + boundary_bonus))

        # Weighted overall (PRD §7.1 domain-specific coefficients)
        overall = T * coeffs['T'] + S * coeffs['S'] + D * coeffs['D'] + C * coeffs['C']
        overall = round(overall, 1)

        # Thresholds — PRD §6 canonical (80/65/50)
        if   overall >= 80: heatmap_class = 'high'
        elif overall >= 65: heatmap_class = 'amber'
        elif overall >= 50: heatmap_class = 'low'
        else:               heatmap_class = 'critical'

        return {
            'overall':              overall,
            'terminology_accuracy': round(T, 1),
            'structural_validity':  round(S, 1),
            'domain_consistency':   round(D, 1),
            'contextual_coherence': round(C, 1),
            'heatmap_class':        heatmap_class,
        }

quality_heuristic_engine = QualityHeuristicEngine()

"""
FTS Retrieval Service — Stage 2.

Retrieves relevant context chunks using PostgreSQL full-text search.
Stage 5 upgrades this to vector similarity search via pgvector.

Domain-to-KB mapping (PRD §7.2):
  code    → "engineering"   (NOT "code")
  medical → "medical"
  legal   → "legal"
"""

from proxima.models.knowledge import DomainKnowledgeChunk

# PRD §7.2 — frozen mapping
DOMAIN_TO_KB = {
    'code':    'engineering',
    'medical': 'medical',
    'legal':   'legal',
}

class FTSRetrievalService:

    async def retrieve(
        self, query: str, domain_weights: dict,
        db: AsyncSession, top_k: int = 3
    ) -> str | None:
        """
        FTS retrieval from domain_knowledge_chunks.
        Uses DOMAIN_TO_KB mapping to resolve kb_name.
        Returns concatenated chunk content or None.
        """
        dominant = max(domain_weights, key=domain_weights.get)
        kb_name  = DOMAIN_TO_KB.get(dominant, dominant)  # fallback to domain name

        stmt = (
            select(DomainKnowledgeChunk)
            .where(DomainKnowledgeChunk.kb_name == kb_name)
            .where(
                func.to_tsvector('english', DomainKnowledgeChunk.content)
                .match(func.plainto_tsquery('english', query[:200]))
            )
            .limit(top_k)
        )
        result = await db.execute(stmt)
        chunks = result.scalars().all()
        if not chunks:
            return None
        return '\n\n---\n\n'.join(c.content for c in chunks)

fts_retrieval = FTSRetrievalService()

"""
Semantic Compare Service — Stage 2 (Temporary implementation).

Pipeline (PRD §7.3):
  Step 1: Chunking (sentence-level, ≤3 sentences)
  Step 2: Jaccard similarity matrix  ← TEMPORARY (Stage 5 replaces with embeddings)
  Step 3: Hungarian alignment
  Step 4: Classification by threshold
  Step 5: LLM analysis for MODIFIED pairs (via ModelRegistry)

Stage 5 replaces Step 2 with embedding cosine similarity.
"""

import re
from scipy.optimize import linear_sum_assignment
import numpy as np

def split_sentences(text: str, max_per_chunk: int = 3) -> list[str]:
    sentences = re.split(r'(?<=[.!?])\s+', text.strip())
    chunks = []
    for i in range(0, len(sentences), max_per_chunk):
        chunk = ' '.join(sentences[i:i + max_per_chunk]).strip()
        if chunk:
            chunks.append(chunk)
    return chunks

def jaccard(a: str, b: str) -> float:
    """Jaccard word-level similarity. Temporary — replaced by cosine in Stage 5."""
    sa = set(a.lower().split())
    sb = set(b.lower().split())
    if not sa and not sb:
        return 1.0
    return len(sa & sb) / len(sa | sb)

def classify_pair(score: float) -> str:
    """PRD §7.3 Step 4 thresholds."""
    if   score >= 0.97: return 'IDENTICAL'
    elif score >= 0.82: return 'MODIFIED'    # possible semantic shift
    elif score >= 0.72: return 'MODIFIED'    # textual change
    elif score >= 0.65: return 'CRITICAL'
    else:               return 'REWRITTEN'

class SemanticCompareService:

    async def compare(
        self, text_a: str, text_b: str, mode: str,
        db: AsyncSession
    ) -> dict:

        # Step 1: Chunk both documents
        chunks_a = split_sentences(text_a)
        chunks_b = split_sentences(text_b)

        if not chunks_a or not chunks_b:
            return {
                'similarity_score': 0.0, 'summary': 'One or both documents are empty.',
                'changes': [], 'critical_count': 0,
                'semantic_count': 0, 'textual_count': 0
            }

        # Step 2: Jaccard similarity matrix (TEMPORARY — Stage 5 replaces)
        n, m = len(chunks_a), len(chunks_b)
        sim_matrix = np.array([
            [jaccard(chunks_a[i], chunks_b[j]) for j in range(m)]
            for i in range(n)
        ])

        # Step 3: Hungarian algorithm alignment
        cost_matrix = 1.0 - sim_matrix
        # Pad to square if needed
        size = max(n, m)
        padded = np.ones((size, size))
        padded[:n, :m] = cost_matrix
        row_ind, col_ind = linear_sum_assignment(padded)

        # Step 4 + 5: Classify and LLM-analyze MODIFIED pairs
        changes = []
        scores_for_avg = []

        for r, c in zip(row_ind, col_ind):
            if r >= n or c >= m:
                continue  # padded rows
            score     = float(sim_matrix[r][c])
            change_type = classify_pair(score)
            scores_for_avg.append(score)
            is_critical = change_type == 'CRITICAL'

            semantic_shift = False
            if change_type == 'MODIFIED' and 0.65 <= score < 0.82:
                # Step 5: LLM analysis
                semantic_shift = await self._llm_semantic_check(
                    chunks_a[r], chunks_b[c], db
                )

            changes.append({
                'type':           change_type,
                'location':       f"Chunk A[{r}] ↔ B[{c}]",
                'original':       chunks_a[r],
                'revised':        chunks_b[c],
                'significance':   round((1.0 - score) * 100),
                'is_critical':    is_critical,
                'semantic_shift': semantic_shift,
            })

        similarity_score = round(float(np.mean(scores_for_avg)) * 100, 1) if scores_for_avg else 0.0

        return {
            'similarity_score': similarity_score,
            'summary':          self._build_summary(changes, similarity_score, mode),
            'changes':          changes,
            'critical_count':   sum(1 for c in changes if c['is_critical']),
            'semantic_count':   sum(1 for c in changes if c['semantic_shift']),
            'textual_count':    sum(1 for c in changes if c['type'] == 'MODIFIED' and not c['semantic_shift']),
        }

    async def _llm_semantic_check(
        self, chunk_a: str, chunk_b: str, db: AsyncSession
    ) -> bool:
        """
        Ask the generation model whether two passages mean the same thing.
        Returns True if they DO NOT mean the same (i.e., is a semantic shift).
        """
        from proxima.services.model_registry import model_registry
        model = await model_registry.get_for_task('compare_documents', None, db)

        prompt = (
            "Do these two passages mean the same thing? "
            "Reply with exactly YES or NO, then one sentence explanation.\n\n"
            f"Passage A: {chunk_a[:400]}\n\n"
            f"Passage B: {chunk_b[:400]}"
        )
        tokens = []
        async for token in model_registry.stream_completion(
            model,
            system_prompt="You are a precise legal/medical/technical document analyst.",
            user_message=prompt,
            temperature=0.1,
            max_tokens=60,
        ):
            tokens.append(token)
        answer = ''.join(tokens).strip().upper()
        return answer.startswith('NO')

    def _build_summary(self, changes: list, score: float, mode: str) -> str:
        total    = len(changes)
        critical = sum(1 for c in changes if c['is_critical'])
        semantic = sum(1 for c in changes if c['semantic_shift'])
        return (
            f"{score:.0f}% similarity detected across {total} matched sections. "
            f"{critical} critical change(s), {semantic} semantic shift(s). "
            f"Mode: {mode.upper()}. "
            "NOTE: Stage 2 uses Jaccard similarity. Stage 5 upgrades to embedding-based comparison."
        )

semantic_compare_service = SemanticCompareService()

    big = b'x' * (21 * 1024 * 1024)
    with pytest.raises(UploadSecurityError, match="20 MB"):
        upload_security.validate(big, 'big.txt', 'text/plain')

def test_rejects_unknown_mime():

    with pytest.raises(UploadSecurityError, match="Archive"):
