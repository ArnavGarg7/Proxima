─── Confidence Color Rule ────────────────────────────────────────────

NEVER hardcode hex values for confidence colors anywhere in frontend code.
ALWAYS use Tailwind CSS classes from globals.css:
  .conf-high     ← maps to PRD threshold 80-100
  .conf-amber    ← maps to PRD threshold 65-79
  .conf-low      ← maps to PRD threshold 50-64
  .conf-critical ← maps to PRD threshold 0-49

The class assignment function:
  function getConfClass(heatmapClass: 'high'|'amber'|'low'|'critical'): string {
    return `conf-${heatmapClass}`;
  }

This receives heatmap_class from the backend QHE — never recalculate
thresholds on the frontend. The backend is the single source of truth.

─── prefers-reduced-motion ──────────────────────────────────────────

Wrap the entire app in MotionConfig in App.tsx:

  import { MotionConfig } from 'framer-motion';

  const prefersReduced = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

  export default function App() {
    return (
      <MotionConfig reducedMotion={prefersReduced ? 'always' : 'never'}>
        {/* rest of app */}
      </MotionConfig>
    );
  }

When prefers-reduced-motion is set:
  - All Framer Motion animations become instant (duration=0)
  - GSAP animations: wrap in if (!prefersReduced) blocks
  - Three.js: still renders but no camera movement on scroll

─── Keyboard Navigation ─────────────────────────────────────────────

All interactive elements must be keyboard accessible:

1. Navbar links: native <a> and <button> elements — tab order correct
2. Domain selector tabs: use role="tablist" + role="tab" + aria-selected
   Arrow keys navigate between tabs:
   const handleKeyDown = (e: KeyboardEvent, index: number) => {
     if (e.key === 'ArrowRight') setActive((index + 1) % 3);
     if (e.key === 'ArrowLeft')  setActive((index + 2) % 3);
   };
3. Accordion: Enter/Space to toggle
4. Modal: Escape to close, focus trap inside modal while open
   import { createFocusTrap } from 'focus-trap'; (install: npm install focus-trap)
5. Dropdown menus: arrow keys navigate, Escape closes

─── Focus Visibility ────────────────────────────────────────────────

Add to globals.css:
  /* Visible focus ring for keyboard users */
  :focus-visible {
    outline: 2px solid #C9A84C;
    outline-offset: 2px;
  }
  /* Remove default outline only when not keyboard navigating */
  :focus:not(:focus-visible) {
    outline: none;
  }

Never use outline: none without the :focus-visible exception.

─── Authorization Rules (PRD §12.4) ─────────────────────────────────

Every /admin/api/* endpoint MUST:
1. Call require_admin dependency (raises 403 for role='user')
2. Log to admin_audit_log BEFORE returning response (not after)
3. Validate no self-escalation on role changes

# middleware/admin_middleware.py

async def require_admin(current_user: User = Depends(get_current_user)) -> User:
    """Dependency for all /admin/api/* endpoints."""
    if current_user.role not in ('admin', 'super_admin'):
        raise HTTPException(403, "Admin access required")
    return current_user

async def require_super_admin(current_user: User = Depends(get_current_user)) -> User:
    """Only super_admin can change roles."""
    if current_user.role != 'super_admin':
        raise HTTPException(403, "Super admin access required for role changes")
    return current_user

async def log_admin_action(
    admin: User, action: str,
    target_type: str | None, target_id: str | None,
    payload: dict | None, request: Request, db: AsyncSession
) -> None:
    """Must be called BEFORE response is returned."""
    entry = AdminAuditLog(
        admin_id=admin.id, action=action,
        target_type=target_type, target_id=str(target_id) if target_id else None,
        payload=payload, ip_address=request.client.host
    )
    db.add(entry)
    await db.commit()

@router.get("/overview")
async def admin_overview(db: AsyncSession = Depends(get_db),
                          redis = Depends(get_redis),
                          admin: User = Depends(require_admin)):
    cache_key = "admin:overview"
    cached = await redis.get(cache_key)
    if cached:
        return json.loads(cached)

    # Total users
    total_users = await db.scalar(select(func.count()).select_from(User))

    # Active users last 30 days
    cutoff = datetime.now(timezone.utc) - timedelta(days=30)
    active_30d = await db.scalar(
        select(func.count()).select_from(User).where(User.last_login >= cutoff)
    )

    # Total AI requests and cost this month
    month_start = datetime.now(timezone.utc).replace(day=1, hour=0, minute=0, second=0, microsecond=0)
    mtd_stats = await db.execute(
        select(
            func.count().label('request_count'),
            func.coalesce(func.sum(text('estimated_cost_usd')), 0).label('total_cost')
        ).select_from(AIRequest).where(AIRequest.created_at >= month_start)
    )
    row = mtd_stats.one()

    # Daily requests last 30 days
    daily = await db.execute(
        select(
            func.date_trunc('day', AIRequest.created_at).label('day'),
            func.count().label('count')
        ).where(AIRequest.created_at >= cutoff)
        .group_by('day').order_by('day')
    )
    daily_data = [{'date': str(r.day.date()), 'count': r.count} for r in daily]

    # Requests by tool type
    by_tool = await db.execute(
        select(AIRequest.request_type, func.count().label('count'))
        .group_by(AIRequest.request_type)
        .order_by(text('count DESC'))
    )

    # Cost by model
    by_model = await db.execute(
        select(AIRequest.model, func.coalesce(func.sum(text('estimated_cost_usd')), 0).label('cost'))
        .where(AIRequest.created_at >= month_start)
        .group_by(AIRequest.model)
        .order_by(text('cost DESC'))
    )

    # Recent admin actions
    recent_logs = await db.execute(
        select(AdminAuditLog, User.email)
        .join(User, AdminAuditLog.admin_id == User.id)
        .order_by(AdminAuditLog.created_at.desc()).limit(20)
    )

    result = {
        'total_users':     total_users,
        'active_users_30d': active_30d,
        'mtd_requests':    row.request_count,
        'mtd_cost_usd':    round(float(row.total_cost), 4),
        'daily_requests':  daily_data,
        'by_tool':         [{'tool': r.request_type, 'count': r.count} for r in by_tool],
        'by_model':        [{'model': r.model, 'cost': round(float(r.cost), 4)} for r in by_model],
        'recent_logs':     [{'admin': r.email, 'action': r.AdminAuditLog.action,
                              'created_at': str(r.AdminAuditLog.created_at)} for r in recent_logs],
    }
    await redis.setex(cache_key, 60, json.dumps(result, default=str))
    return result

# ── GET /admin/api/users ─────────────────────────────────────────
@router.get("/users")
async def list_users(
    search: str = None, plan: str = None, role: str = None,
    is_active: bool = None, page: int = 1, limit: int = 20,
    db: AsyncSession = Depends(get_db), admin: User = Depends(require_admin)
):
    stmt = select(User)
    if search:
        stmt = stmt.where(or_(User.email.ilike(f'%{search}%'),
                               User.display_name.ilike(f'%{search}%')))
    if plan:     stmt = stmt.where(User.plan == plan)
    if role:     stmt = stmt.where(User.role == role)
    if is_active is not None: stmt = stmt.where(User.is_active == is_active)
    stmt = stmt.order_by(User.created_at.desc()).offset((page-1)*limit).limit(limit)
    result = await db.execute(stmt)
    users = result.scalars().all()
    total = await db.scalar(select(func.count()).select_from(User))
    return {'users': [u.__dict__ for u in users], 'total': total, 'page': page, 'limit': limit}

# ── PATCH /admin/api/users/:id/plan ─────────────────────────────
@router.patch("/users/{user_id}/plan")
async def update_user_plan(
    user_id: UUID, body: dict,
    request: Request, db: AsyncSession = Depends(get_db),
    admin: User = Depends(require_admin)
):
    plan = body.get('plan')
    if plan not in ('free', 'pro', 'enterprise'):
        raise HTTPException(400, f"Invalid plan: {plan}")
    user = await db.get(User, user_id)
    if not user: raise HTTPException(404, "User not found")
    old_plan = user.plan
    user.plan = plan
    # Log BEFORE commit
    await log_admin_action(admin, 'update_user_plan', 'user', str(user_id),
                            {'from': old_plan, 'to': plan}, request, db)
    await db.commit()
    return {'ok': True, 'user_id': str(user_id), 'plan': plan}

# ── GET /admin/api/prompts/:key/diff ────────────────────────────
@router.get("/prompts/{key:path}/diff")
async def diff_prompt_versions(
    key: str, v1: int, v2: int,
    db: AsyncSession = Depends(get_db), admin=Depends(require_admin)
):
    from proxima.services.prompt_registry import CANONICAL_PROMPT_KEYS
    if key not in CANONICAL_PROMPT_KEYS:
        raise HTTPException(400, f"Unknown prompt key: {key}")
    pv1 = await db.scalar(select(PromptVersion).where(
        PromptVersion.prompt_key == key, PromptVersion.version == v1))
    pv2 = await db.scalar(select(PromptVersion).where(
        PromptVersion.prompt_key == key, PromptVersion.version == v2))
    if not pv1 or not pv2:
        raise HTTPException(404, "One or both versions not found")
    import difflib
    diff = list(difflib.unified_diff(
        pv1.content.splitlines(keepends=True),
        pv2.content.splitlines(keepends=True),
        fromfile=f"v{v1}", tofile=f"v{v2}"
    ))
    return {'key': key, 'v1': v1, 'v2': v2, 'diff': ''.join(diff)}

        'total': total, 'page': page, 'limit': limit,
        'kb_stats': [{'kb_name': r.kb_name, 'total': r.total, 'embedded': r.embedded}

    }

# ── POST /admin/api/knowledge/chunks ────────────────────────────
@router.post("/knowledge/chunks")
async def add_knowledge_chunk(
    body: dict, request: Request,
    db: AsyncSession = Depends(get_db), admin: User = Depends(require_admin)
):
    kb_name = body.get('kb_name')
    content = body.get('content', '').strip()
    if kb_name not in ('engineering', 'medical', 'legal'):
        raise HTTPException(400, f"kb_name must be one of: engineering, medical, legal")
    if not content:
        raise HTTPException(400, "content is required")
    # Get next chunk_index
    max_idx = await db.scalar(
        select(func.max(DomainKnowledgeChunk.chunk_index))
        .where(DomainKnowledgeChunk.kb_name == kb_name)
    ) or 0
    chunk = DomainKnowledgeChunk(
        kb_name=kb_name, title=body.get('title'),
        content=content, source=body.get('source'),
        chunk_index=max_idx + 1,
        token_count=len(content.split()),
    )
    db.add(chunk)
    await log_admin_action(admin, 'add_knowledge_chunk', 'knowledge', kb_name,
                            {'title': body.get('title'), 'chars': len(content)}, request, db)
    await db.commit()
    return {'ok': True, 'id': str(chunk.id)}

# ── POST /admin/api/knowledge/embed ─────────────────────────────
@router.post("/knowledge/embed")
async def trigger_kb_embedding(
    request: Request, body: dict = {},
    db: AsyncSession = Depends(get_db), admin: User = Depends(require_admin)
):
    from proxima.celery_app import app as celery_app
    kb_name = body.get('kb_name')  # None = all KBs
    # Count chunks needing embedding
    stmt = select(func.count()).select_from(DomainKnowledgeChunk).where(
        DomainKnowledgeChunk.embedding == None)
    if kb_name:
        stmt = stmt.where(DomainKnowledgeChunk.kb_name == kb_name)
    pending_count = await db.scalar(stmt)
    # Queue Celery task
    task = celery_app.send_task(
        'proxima.tasks.embeddings.embed_knowledge_base',
        kwargs={'kb_name': kb_name},
        queue='proxima:embed'
    )
    await log_admin_action(admin, 'trigger_kb_embedding', 'knowledge', kb_name or 'all',
                            {'pending_chunks': pending_count, 'celery_id': task.id}, request, db)
    return {'ok': True, 'celery_id': task.id, 'pending_chunks': pending_count}

        select(func.coalesce(func.sum(text('estimated_cost_usd')), 0))
        .select_from(AIRequest).where(AIRequest.created_at >= month_start)
    )

    # Daily cost last 30 days
    daily = await db.execute(
        select(func.date_trunc('day', AIRequest.created_at).label('day'),
               func.coalesce(func.sum(text('estimated_cost_usd')), 0).label('cost'))
        .where(AIRequest.created_at >= cutoff_30d)
        .group_by('day').order_by('day')
    )

    # Cost by model
    by_model = await db.execute(
        select(AIRequest.model,
               func.coalesce(func.sum(text('estimated_cost_usd')), 0).label('cost'),
               func.count().label('requests'))
        .where(AIRequest.created_at >= month_start)
        .group_by(AIRequest.model).order_by(text('cost DESC'))
    )

    # Cost by tool
    by_tool = await db.execute(
        select(AIRequest.request_type,
               func.coalesce(func.sum(text('estimated_cost_usd')), 0).label('cost'))
        .where(AIRequest.created_at >= month_start)
        .group_by(AIRequest.request_type).order_by(text('cost DESC'))
    )

    from proxima.config import settings
    budget     = settings.monthly_ai_budget_usd
    utilization = round(float(mtd_cost) / budget * 100, 1) if budget > 0 else 0

    result = {
        'mtd_cost_usd':  round(float(mtd_cost), 4),
        'budget_usd':    budget,
        'utilization_pct': utilization,
        'over_threshold': utilization >= settings.cost_alert_threshold * 100,
        'daily':  [{'date': str(r.day.date()), 'cost': round(float(r.cost), 4)} for r in daily],
        'by_model': [{'model': r.model, 'cost': round(float(r.cost), 4),
                      'requests': r.requests} for r in by_model],
        'by_tool':  [{'tool': r.request_type, 'cost': round(float(r.cost), 4)} for r in by_tool],
    }
    await redis.setex(cache_key, 300, json.dumps(result, default=str))
    return result

import { describe, it, expect } from 'vitest';

function getVertices(weights: {code: number, medical: number, legal: number},
                     radius: number, cx: number, cy: number) {
  const angles = { legal: -90, code: 30, medical: 150 };
  return Object.entries(angles).map(([domain, angle]) => {
    const r = (weights[domain as keyof typeof weights] ?? 0) * radius;
    const rad = (angle * Math.PI) / 180;
    return { x: cx + r * Math.cos(rad), y: cy + r * Math.sin(rad) };
  });
}

describe('DomainRadar vertex calculation', () => {
  it('places equal weights as equilateral triangle', () => {
    const v = getVertices({code:0.33, medical:0.33, legal:0.34}, 100, 110, 110);
    expect(v).toHaveLength(3);
    v.forEach(pt => {
      expect(typeof pt.x).toBe('number');
      expect(typeof pt.y).toBe('number');
    });
  });

  it('places 100% legal at top-center', () => {
    const v = getVertices({code:0, medical:0, legal:1}, 100, 110, 110);
    const legalPoint = v.find((_, i) => i === 0); // legal is index 0 in angles
    expect(legalPoint).toBeDefined();
  });

  it('places 0% domain at center', () => {
    const v = getVertices({code:0, medical:0, legal:0}, 100, 110, 110);
    v.forEach(pt => {
      expect(Math.abs(pt.x - 110)).toBeLessThan(1);
      expect(Math.abs(pt.y - 110)).toBeLessThan(1);
    });
  });
});
