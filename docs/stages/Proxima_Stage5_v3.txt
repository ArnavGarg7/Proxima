"""
EmbeddingService: all embedding calls go through ModelRegistry.
Never import google.generativeai or any provider SDK directly here.
"""

import hashlib
import json
from proxima.services.model_registry import model_registry
from proxima.database import get_db, get_redis
from proxima.config import settings

class EmbeddingService:

    def _cache_key(self, text: str, model_id: str, version: str,
                   dims: int, task_type: str) -> str:
        """
        Versioned cache key — PRD fix: includes model+version+dims+task.
        NOT just md5(text) — that would serve stale embeddings after model change.
        """
        fingerprint = f"{text}||{model_id}||{version}||{dims}||{task_type}"
        return f"embed:{hashlib.md5(fingerprint.encode()).hexdigest()}"

    async def embed_text(
        self, text: str, db, redis,
        task_type: str = 'RETRIEVAL_DOCUMENT'
    ) -> tuple[list[float], dict]:
        """
        Embed a single text string.
        Returns (embedding_vector, embedding_meta) where meta contains
        the versioning fields to store on the DB record.
        """
        model = await model_registry.get_default_embedding(db)
        cache_key = self._cache_key(
            text, model.model_id,
            model.embedding_version or '1.0',
            model.embedding_dimensions or 768,
            task_type
        )

        # Check cache
        cached = await redis.get(cache_key)
        if cached:
            return json.loads(cached), self._meta(model)

        # Call via ModelRegistry
        embedding = await self._call_embed(model, [text], task_type)
        result = embedding[0]

        # Cache with versioned key
        await redis.setex(cache_key, settings.embedding_cache_ttl, json.dumps(result))
        return result, self._meta(model)

    async def embed_batch(
        self, texts: list[str], db, redis,
        task_type: str = 'RETRIEVAL_DOCUMENT'
    ) -> tuple[list[list[float]], dict]:
        """
        Batch embed up to 100 texts.
        Returns (list_of_vectors, embedding_meta).
        """
        if not texts:
            return [], {}
        model = await model_registry.get_default_embedding(db)
        embeddings = await self._call_embed(model, texts, task_type)
        return embeddings, self._meta(model)

    async def _call_embed(
        self, model, texts: list[str], task_type: str
    ) -> list[list[float]]:
        """Route to correct provider via ModelRegistry."""
        client = await model_registry.build_embedding_client(model)

        if model.provider == 'google':
            result = client.embed_content(
                model=model.model_id,
                content=texts if len(texts) > 1 else texts[0],
                task_type=task_type,
            )
            raw = result['embedding']
            # Google returns list[float] for single, list[list[float]] for batch
            if isinstance(raw[0], float):
                return [raw]
            return raw
        else:
            raise ValueError(f"Unsupported embedding provider: {model.provider}")

    def _meta(self, model) -> dict:
        """Returns the versioning fields to store on each embedded record."""
        from datetime import datetime, timezone
        return {
            'embedding_model':      model.model_id,
            'embedding_version':    model.embedding_version or '1.0',
            'embedding_dimensions': model.embedding_dimensions or 768,
            'embedded_at':          datetime.now(timezone.utc),
        }

embedding_service = EmbeddingService()

@app.task(name='proxima.tasks.embeddings.embed_document_chunks',
          queue='proxima:embed', bind=True, max_retries=2, default_retry_delay=60)
def embed_document_chunks(self, document_id: str):
    """
    Embed all NULL-embedding chunks for a document.
    Populates embedding_model, embedding_version, embedding_dimensions, embedded_at.
    """
    import asyncio
    from proxima.database import AsyncSessionLocal, get_redis
    from proxima.models.document import Document, DocumentChunk
    from proxima.services.embedding_service import embedding_service

    async def _run():
        redis = await get_redis()
        async with AsyncSessionLocal() as db:
            stmt = (select(DocumentChunk)
                    .where(DocumentChunk.document_id == UUID(document_id),
                           DocumentChunk.embedding == None)
                    .order_by(DocumentChunk.chunk_index))
            chunks = (await db.execute(stmt)).scalars().all()
            if not chunks:
                return

            BATCH = 50
            for i in range(0, len(chunks), BATCH):
                batch = chunks[i:i + BATCH]
                texts = [c.content for c in batch]
                embeddings, meta = await embedding_service.embed_batch(texts, db, redis)
                for chunk, emb in zip(batch, embeddings):
                    chunk.embedding           = emb
                    chunk.embedding_model     = meta['embedding_model']
                    chunk.embedding_version   = meta['embedding_version']
                    chunk.embedding_dimensions = meta['embedding_dimensions']
                    chunk.embedded_at         = meta['embedded_at']
                await db.commit()

            # Update document ingest status
            doc = await db.get(Document, UUID(document_id))
            if doc:
                doc.ingest_status = 'EMBEDDED'
                await db.commit()

    try:
        asyncio.run(_run())
    except Exception as e:
        raise self.retry(exc=e)

"""
Vector Retrieval Service — Stage 5 upgrade.

Replaces FTS Retrieval (Stage 2) with pgvector cosine similarity.
Falls back to FTS if embeddings not yet available.

Domain-to-KB mapping (PRD §7.2 — unchanged):
  code    → "engineering"
  medical → "medical"
  legal   → "legal"
"""

from pgvector.sqlalchemy import cosine_distance   # official pgvector comparator

DOMAIN_TO_KB = {
    'code':    'engineering',
    'medical': 'medical',
    'legal':   'legal',
}

class VectorRetrievalService:

    async def retrieve(
        self, query: str, domain_weights: dict,
        db: AsyncSession, redis, top_k: int = 3
    ) -> str | None:
        """
        Primary: pgvector cosine similarity.
        Fallback: FTS (if no embeddings available or embed fails).
        """
        dominant = max(domain_weights, key=domain_weights.get)
        kb_name  = DOMAIN_TO_KB.get(dominant, dominant)

        # Try vector search
        try:
            query_vec, _ = await embedding_service.embed_text(
                query[:500], db, redis, task_type='RETRIEVAL_QUERY'
            )
            chunks = await self._vector_search(query_vec, kb_name, db, top_k)
            if chunks:
                return '\n\n---\n\n'.join(c.content for c in chunks)
        except Exception:
            pass  # fall through to FTS

        # FTS fallback
        return await self._fts_search(query, kb_name, db, top_k)

    async def _vector_search(
        self, embedding: list[float], kb_name: str,
        db: AsyncSession, top_k: int
    ) -> list:
        """
        pgvector cosine distance search — official comparator syntax.
        cosine_distance from pgvector.sqlalchemy is the correct import.
        Lower distance = more similar.
        """
        stmt = (
            select(DomainKnowledgeChunk)
            .where(DomainKnowledgeChunk.kb_name == kb_name)
            .where(DomainKnowledgeChunk.embedding != None)
            .order_by(cosine_distance(DomainKnowledgeChunk.embedding, embedding))
            .limit(top_k)
        )
        result = await db.execute(stmt)
        return result.scalars().all()

    async def _fts_search(
        self, query: str, kb_name: str,
        db: AsyncSession, top_k: int
    ) -> str | None:
        stmt = (
            select(DomainKnowledgeChunk)
            .where(DomainKnowledgeChunk.kb_name == kb_name)
            .where(
                func.to_tsvector('english', DomainKnowledgeChunk.content)
                .match(func.plainto_tsquery('english', query[:200]))
            )
            .limit(top_k)
        )
        result = await db.execute(stmt)
        chunks = result.scalars().all()
        if not chunks:
            return None
        return '\n\n---\n\n'.join(c.content for c in chunks)

vector_retrieval = VectorRetrievalService()

"""
Semantic Compare Service — Stage 5 (Production implementation).

Pipeline (PRD §7.3):
  Step 1: Chunking (sentence-level, ≤3 sentences) — unchanged
  Step 2: Embedding cosine similarity matrix  ← REPLACES Jaccard
  Step 3: Hungarian alignment — unchanged
  Step 4: Classification by threshold — unchanged
  Step 5: LLM analysis for MODIFIED pairs — unchanged
"""

from scipy.optimize import linear_sum_assignment
import numpy as np

def split_sentences(text: str, max_per_chunk: int = 3) -> list[str]:
    import re
    sentences = re.split(r'(?<=[.!?])\s+', text.strip())
    chunks = []
    for i in range(0, len(sentences), max_per_chunk):
        chunk = ' '.join(sentences[i:i + max_per_chunk]).strip()
        if chunk:
            chunks.append(chunk)
    return chunks

def classify_pair(score: float) -> str:
    """PRD §7.3 Step 4 — exact thresholds, unchanged."""
    if   score >= 0.97: return 'IDENTICAL'
    elif score >= 0.82: return 'MODIFIED'
    elif score >= 0.72: return 'MODIFIED'
    elif score >= 0.65: return 'CRITICAL'
    else:               return 'REWRITTEN'

def cosine_similarity(a: list[float], b: list[float]) -> float:
    """Pure numpy cosine similarity between two embedding vectors."""
    va = np.array(a, dtype=np.float32)
    vb = np.array(b, dtype=np.float32)
    norm_a = np.linalg.norm(va)
    norm_b = np.linalg.norm(vb)
    if norm_a < 1e-10 or norm_b < 1e-10:
        return 0.0
    return float(np.dot(va, vb) / (norm_a * norm_b))

class SemanticCompareService:

    async def compare(
        self, text_a: str, text_b: str, mode: str,
        db: AsyncSession, redis
    ) -> dict:

        # Step 1: Chunk both documents
        chunks_a = split_sentences(text_a)
        chunks_b = split_sentences(text_b)

        if not chunks_a or not chunks_b:
            return {
                'similarity_score': 0.0,
                'summary': 'One or both documents are empty.',
                'changes': [], 'critical_count': 0,
                'semantic_count': 0, 'textual_count': 0,
            }

        # Step 2: Embedding similarity matrix (PRODUCTION — replaces Jaccard)
        all_texts  = chunks_a + chunks_b
        all_embeds, _ = await embedding_service.embed_batch(
            all_texts, db, redis, task_type='SEMANTIC_SIMILARITY'
        )
        embeds_a = all_embeds[:len(chunks_a)]
        embeds_b = all_embeds[len(chunks_a):]

        n, m = len(chunks_a), len(chunks_b)
        sim_matrix = np.array([
            [cosine_similarity(embeds_a[i], embeds_b[j]) for j in range(m)]
            for i in range(n)
        ])

        # Step 3: Hungarian algorithm alignment
        size   = max(n, m)
        padded = np.ones((size, size))
        padded[:n, :m] = 1.0 - sim_matrix   # cost = 1 - similarity
        row_ind, col_ind = linear_sum_assignment(padded)

        # Step 4 + 5: Classify and LLM-analyze
        changes, scores_for_avg = [], []

        for r, c in zip(row_ind, col_ind):
            if r >= n or c >= m:
                continue
            score       = float(sim_matrix[r][c])
            change_type = classify_pair(score)
            scores_for_avg.append(score)
            is_critical = change_type == 'CRITICAL'

            semantic_shift = False
            if change_type == 'MODIFIED' and 0.65 <= score < 0.82:
                semantic_shift = await self._llm_semantic_check(
                    chunks_a[r], chunks_b[c], db
                )

            changes.append({
                'type':           change_type,
                'location':       f"Chunk A[{r}] ↔ B[{c}]",
                'original':       chunks_a[r],
                'revised':        chunks_b[c],
                'significance':   round((1.0 - score) * 100),
                'is_critical':    is_critical,
                'semantic_shift': semantic_shift,
            })

        similarity_score = round(float(np.mean(scores_for_avg)) * 100, 1) if scores_for_avg else 0.0

        return {
            'similarity_score': similarity_score,
            'summary': self._build_summary(changes, similarity_score, mode),
            'changes': changes,
            'critical_count':  sum(1 for c in changes if c['is_critical']),
            'semantic_count':  sum(1 for c in changes if c['semantic_shift']),
            'textual_count':   sum(1 for c in changes if
                                   c['type'] == 'MODIFIED' and not c['semantic_shift']),
        }

    async def _llm_semantic_check(
        self, chunk_a: str, chunk_b: str, db: AsyncSession
    ) -> bool:
        model = await model_registry.get_for_task('compare_documents', None, db)
        prompt = (
            "Do these two passages mean the same thing? "
            "Reply YES or NO only.\n\n"
            f"A: {chunk_a[:400]}\n\nB: {chunk_b[:400]}"
        )
        tokens = []
        async for token in model_registry.stream_completion(
            model, "You are a precise document analyst.", prompt, 0.0, 5
        ):
            tokens.append(token)
        return ''.join(tokens).strip().upper().startswith('NO')

    def _build_summary(self, changes: list, score: float, mode: str) -> str:
        total    = len(changes)
        critical = sum(1 for c in changes if c['is_critical'])
        semantic = sum(1 for c in changes if c['semantic_shift'])
        return (
            f"{score:.0f}% similarity across {total} matched sections. "
            f"{critical} critical, {semantic} semantic shift(s). Mode: {mode.upper()}."
        )

semantic_compare_service = SemanticCompareService()

        def r(para, text, bold=False, size=11, color=None):
            run = para.add_run(bleach.clean(str(text), tags=[], strip=True))
            run.bold = bold; run.font.size = Pt(size); run.font.name = 'Georgia'
            if color: run.font.color.rgb = RGBColor(*color)
            return run

        r(doc.add_paragraph(), 'PROXIMA', bold=True, size=18)
        r(doc.add_paragraph(), '─' * 65, size=8, color=(0xC9, 0xA8, 0x4C))
        h = doc.add_heading(safe_title, level=1)
        if h.runs: h.runs[0].font.color.rgb = RGBColor(0x0A, 0x0A, 0x0A)
        meta_str = (
            f"Domain: {metadata.get('domain', '').upper()} · "
            f"Tool: {metadata.get('tool', '')} · "
            f"Generated: {datetime.now().strftime('%B %d, %Y')}"
        )
        if metadata.get('confidence_score'):
            meta_str += f" · Confidence: {metadata['confidence_score']}%"
        r(doc.add_paragraph(), bleach.clean(meta_str, tags=[], strip=True),
          size=9, color=(0x88, 0x88, 0x80))
        doc.add_paragraph()

        for line in safe_content.split('\n'):
            s = line.strip()
            if not s: doc.add_paragraph(); continue
            if s.startswith('### '): doc.add_heading(s[4:], level=3)
            elif s.startswith('## '): doc.add_heading(s[3:], level=2)
            elif s.startswith('# '):  doc.add_heading(s[2:], level=1)
            elif s.startswith(('- ', '* ')): doc.add_paragraph(s[2:], style='List Bullet')
            else: doc.add_paragraph(s)

        foot = doc.add_paragraph()
        r(foot, 'Generated by Proxima · Engineered by Arnav Garg', size=8, color=(0x88,0x88,0x80))
        foot.alignment = WD_ALIGN_PARAGRAPH.CENTER

        buf = io.BytesIO(); doc.save(buf); buf.seek(0)
        return buf.getvalue()

    # ── MARKDOWN ────────────────────────────────────────────────────

    def generate_markdown(self, content: str, title: str, metadata: dict) -> bytes:
        safe_content = bleach.clean(content, tags=[], strip=True)
        safe_title   = bleach.clean(title,   tags=[], strip=True)
        conf = f" · Confidence: {metadata['confidence_score']}%" if metadata.get('confidence_score') else ""
        header = (
            f"# {safe_title}\n\n"
            f"> **Proxima** · Domain: {metadata.get('domain','').upper()}"
            f" · {datetime.now().strftime('%B %d, %Y')}{conf}\n"
            f"> Engineered by Arnav Garg\n\n---\n\n"
        )
        return (header + safe_content + "\n\n---\n*Proxima AI Writing Suite*\n").encode('utf-8')

    # ── CSV ──────────────────────────────────────────────────────────

    def generate_csv(self, segment_scores: list[dict]) -> bytes:
        import csv, io
        out    = io.StringIO()
        fields = ['segment_index','segment_text','overall','terminology_accuracy',
                  'structural_validity','domain_consistency','contextual_coherence','heatmap_class']
        w = csv.DictWriter(out, fieldnames=fields, extrasaction='ignore')
        w.writeheader()
        # Sanitize text fields in each row
        for row in segment_scores:
            row['segment_text'] = bleach.clean(str(row.get('segment_text', '')), tags=[], strip=True)
            w.writerow(row)
        return out.getvalue().encode('utf-8')

    # ── JSON ──────────────────────────────────────────────────────────

    def generate_json(self, document: dict, metadata: dict) -> bytes:
        import json
        # Sanitize any string fields in document
        safe_doc = {k: bleach.clean(str(v), tags=[], strip=True) if isinstance(v, str) else v
                    for k, v in document.items()}
        return json.dumps({
            'proxima_export': True, 'version': '4.0',
            'generated_at':   datetime.now().isoformat(),
            'engineer':       'Arnav Garg',
            'document':       safe_doc,
            'metadata':       metadata,
        }, indent=2, default=str).encode('utf-8')

    # ── ZIP ───────────────────────────────────────────────────────────

    def generate_zip(self, files: list[tuple[str, bytes]]) -> bytes:
        import zipfile, io
        buf = io.BytesIO()
        with zipfile.ZipFile(buf, 'w', zipfile.ZIP_DEFLATED) as zf:
            for name, data in files:
                # Sanitize filename
                safe_name = bleach.clean(name, tags=[], strip=True).replace('/', '_')
                zf.writestr(safe_name, data)
        buf.seek(0)
        return buf.getvalue()

@app.task(name='proxima.tasks.exports.generate_export',
          queue='proxima:exports', bind=True, max_retries=2, default_retry_delay=30)
def generate_export(self, export_id: str):
    import asyncio
    from proxima.database import AsyncSessionLocal
    from proxima.models.export import Export
    from proxima.services.export_service import export_service
    from proxima.services.storage_service import storage_service

    async def _run():
        async with AsyncSessionLocal() as db:
            export = await db.get(Export, UUID(export_id))
            if not export:
                return
            try:
                export.status = 'processing'
                await db.commit()

                meta    = export.metadata or {}
                content = meta.get('content', '')
                title   = meta.get('title', 'Proxima Export')
                doc_meta = meta.get('doc_metadata', {})

                if export.format == 'pdf':
                    data = export_service.generate_pdf(content, title, doc_meta)
                    ct   = 'application/pdf'; ext = 'pdf'
                elif export.format == 'docx':
                    data = export_service.generate_docx(content, title, doc_meta)
                    ct   = 'application/vnd.openxmlformats-officedocument.wordprocessingml.document'
                    ext  = 'docx'
                elif export.format == 'md':
                    data = export_service.generate_markdown(content, title, doc_meta)
                    ct   = 'text/markdown'; ext = 'md'
                elif export.format == 'csv':
                    data = export_service.generate_csv(meta.get('segment_scores', []))
                    ct   = 'text/csv'; ext = 'csv'
                elif export.format == 'json':
                    data = export_service.generate_json({'content': content}, doc_meta)
                    ct   = 'application/json'; ext = 'json'
                else:
                    raise ValueError(f"Unknown format: {export.format}")

                key = storage_service.export_key(str(export.user_id), export_id, ext)
                await storage_service.upload_bytes(data, key, ct)
                url = storage_service.get_presigned_download(key, expires=604800)

                from datetime import datetime, timedelta, timezone
                export.status     = 'ready'
                export.file_url   = url
                export.file_name  = f"{title[:50].lower().replace(' ','_')}.{ext}"
                export.file_size  = len(data)
                export.expires_at = datetime.now(timezone.utc) + timedelta(days=7)
                await db.commit()

            except Exception as e:
                export.status        = 'failed'
                export.error_message = str(e)
                await db.commit()
                raise self.retry(exc=e)

    try:
        asyncio.run(_run())
    except Exception as e:
        raise self.retry(exc=e)

# Fields to redact in ALL locations (request body, nested JSON, headers, task args)
REDACT_FIELDS = {
    'text', 'content', 'original_text', 'current_text',
    'bullet_points', 'code', 'error_message', 'input_bullets',
    'email', 'password', 'token', 'access_token', 'refresh_token',
    'authorization', 'cookie', 'set-cookie', 'x-api-key',
}

def _redact_dict(obj: dict | list | str, depth: int = 0) -> dict | list | str:
    """Recursively redact sensitive fields from any nested structure."""
    if depth > 8:  # prevent infinite recursion
        return '[MAX_DEPTH]'
    if isinstance(obj, dict):
        return {
            k: '[REDACTED]' if k.lower() in REDACT_FIELDS
               else _redact_dict(v, depth + 1)
            for k, v in obj.items()
        }
    if isinstance(obj, list):
        return [_redact_dict(item, depth + 1) for item in obj]
    if isinstance(obj, str) and len(obj) > 500:
        return '[TRUNCATED]'  # don't send large strings
    return obj

def _redact_url(url: str) -> str:
    """Remove query string from URLs (may contain tokens)."""
    return url.split('?')[0] if url else url

def scrub_event(event: dict, hint: dict) -> dict:
    """
    Comprehensive PII and sensitive data scrubber.
    Covers: request body, headers, nested JSON, task args, breadcrumbs, URLs.
    """
    # 1. Request body
    if 'request' in event:
        req = event['request']
        if 'data' in req:
            req['data'] = _redact_dict(req['data'])
        # 2. Request headers
        if 'headers' in req:
            req['headers'] = {
                k: '[REDACTED]' if k.lower() in REDACT_FIELDS else v
                for k, v in req.get('headers', {}).items()
            }
        # 3. URL query string
        if 'url' in req:
            req['url'] = _redact_url(req['url'])
        if 'query_string' in req:
            req['query_string'] = '[REDACTED]'

    # 4. User object (never send email)
    if 'user' in event:
        user = event['user']
        user['email']    = '[REDACTED]'
        user['username'] = '[REDACTED]'
        user['ip_address'] = '[REDACTED]'

    # 5. Celery task args and kwargs
    if 'extra' in event:
        extra = event.get('extra', {})
        for key in ('celery-job', 'task_args', 'task_kwargs'):
            if key in extra:
                extra[key] = '[REDACTED]'

    # 6. Exception values (may contain stack locals with content)
    if 'exception' in event:
        for exc in event['exception'].get('values', []):
            frames = exc.get('stacktrace', {}).get('frames', [])
            for frame in frames:
                if 'vars' in frame:
                    frame['vars'] = _redact_dict(frame['vars'])

    # 7. Breadcrumbs (may contain request data)
    if 'breadcrumbs' in event:
        for crumb in event['breadcrumbs'].get('values', []):
            if 'data' in crumb:
                crumb['data'] = _redact_dict(crumb['data'])
            if 'message' in crumb and len(crumb['message']) > 200:
                crumb['message'] = crumb['message'][:200] + '...'

    return event

export function identifyUser(userId: string, plan: string) {
  // Anonymized — hashed ID, no email
  const hash = Math.abs(
    userId.split('').reduce((h, c) => (((h << 5) - h) + c.charCodeAt(0)) | 0, 0)
  );
  posthog.identify(`user_${hash}`, { plan });
}

## Architecture
```
React 18 + Three.js + GSAP (Frontend)
         ↓ HTTPS + SSE
FastAPI + RS256 JWT (API Gateway)
         ↓
ModelRegistry → Gemini 2.0 Flash / text-embedding-004
         ↓
PostgreSQL 16 + pgvector | Redis 7 | S3/R2
         ↑
```

### Prerequisites
Node.js 20+, Python 3.11+, PostgreSQL 16+ with pgvector, Redis 7+

### Quick Start
```bash
# 1. Generate RSA keys for RS256 JWT
cd backend && python scripts/generate_keys.py

alembic upgrade head
python scripts/seed_templates_full.py
python scripts/seed_knowledge_base.py
uvicorn proxima.main:app --reload --port 8000

# 3. Celery workers (separate terminals)
celery -A proxima.celery_app worker -Q proxima:default -c 2 --loglevel=info
celery -A proxima.celery_app worker -Q proxima:ai     -c 2 --loglevel=info
celery -A proxima.celery_app worker -Q proxima:exports -c 1 --loglevel=info
celery -A proxima.celery_app worker -Q proxima:embed  -c 1 --loglevel=info

# 4. Embed knowledge base (one-time, after workers start)
celery -A proxima.celery_app call proxima.tasks.embeddings.embed_knowledge_base

# 5. Frontend
cd frontend && cp .env.example .env && npm install && npm run dev

cd backend && pytest tests/ -v --cov=proxima
cd frontend && npx vitest run && npx playwright test
```
